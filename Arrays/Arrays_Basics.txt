7.
Given two sorted arrays, merge into third array

n=int(input())
m=int(input())
a=list(map(int,input().strip().split()))[:n]
b=list(map(int,input().strip().split()))[:m]
c=[]
for i,j in zip(a,b):
	c.append(i)
	c.append(j)

return c
